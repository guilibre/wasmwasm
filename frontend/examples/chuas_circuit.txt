in1 = delay 1 {x. 0.1}
in2 = delay 1 {x. 0}
in3 = delay 1 {x. 0}

alpha = 15.7
beta = 28
m0 = -1.143
m1 = -0.714
dt = 0.02

abs = {x. x * sign x}

h = {x.
  tmp = abs (x + 1)
  tmp = tmp - abs (x - 1)
  tmp = tmp * (m0 - m1)
  tmp = tmp * 0.5
  tmp + m1 * x
}

x = @in1
y = @in2
z = @in3

xdot = alpha * (y - x - h x)
ydot = z + x - y
zdot = beta * -y

in1 <- x + dt * xdot
in2 <- y + dt * ydot
in3 <- z + dt * zdot

#always clip so your ears don't bleed
OUT <- 0.2 * clip (0.2 * x)
